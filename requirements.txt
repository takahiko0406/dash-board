name: Conditional RMB Dashboard Monitor

on:
  workflow_dispatch:
  schedule:
    # Runs Monday-Friday at 22:30 UTC = 07:30 Japan time next day.
    # Adjust if you want a different time.
    - cron: "30 22 * * 1-5"

jobs:
  run-rmb-dashboard:
    runs-on: ubuntu-latest
    timeout-minutes: 60

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Run Conditional RMB dashboard and send Telegram
        env:
          BOT_TOKEN: ${{ secrets.BOT_TOKEN }}
          CHAT_ID: ${{ secrets.CHAT_ID }}
        run: |
          python run_conditional_rmb_dashboard_telegram.py

      - name: Upload dashboard outputs
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: conditional-rmb-dashboard-outputs
          path: |
            *.csv
            *.txt
            *_log.txt
